#include <stdio.h>
#include <math.h>
#include <windows.h>

int main()
{
    int b = 0, c = 0;
    int i, j;
    printf("input b\n");
    scanf_s("%d", &b);
    if (b != 0)
        printf("input c\n");
    scanf_s("%d", &c);
    if (b != 0)
        printf("ok\n");
    for (i = -999; i < 999; i = i + 1)
    {
        //printf("first circle\n");
        for (j = -999; j < 999; j = j + 1)
        {
            //printf("j = %d\n",j);
            if (i + j == b) {
                //printf("1\n");
                if (i * j == c)
                    printf("(x+%d)(x+%d)\n", i, j);
            }
        }
    }
    system("pause");
}
